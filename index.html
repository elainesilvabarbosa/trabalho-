        public class Matriz3D {
            public static void main(String[] args) {
                // Definindo as dimensões da matriz 3D
                int x = 3; // profundidade
                int y = 3; // linhas
                int z = 3; // colunas
        
                // Criando a matriz 3D
                int[][][] matriz = new int[x][y][z];
        
                // Preenchendo a matriz com valores
                for (int i = 0; i < x; i++) {
                    for (int j = 0; j < y; j++) {
                        for (int k = 0; k < z; k++) {
                            matriz[i][j][k] = i + j + k; // exemplo de preenchimento
                        }
                    }
                }
        
                // Imprimindo a matriz 3D
                for (int i = 0; i < x; i++) {
                    System.out.println("Camada " + i + ":");
                    for (int j = 0; j < y; j++) {
                        for (int k = 0; k < z; k++) {
                            System.out.print(matriz[i][j][k] + " ");
                        }
                        System.out.println();
                    }
                    System.out.println();
                }
            }
        }
